# ctx-tree v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:test-driven-development` for every implementation task — strict red→green→refactor. Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to execute task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build ctx-tree v1 — a TypeScript MCP server that exposes surgical recall tools (`ctx-tree.read`, `ctx-tree.grep`, `ctx-tree.compose`, `ctx-tree.search`, `ctx-tree.neighbors`, `ctx-tree.path_to_root`, `ctx-tree.recent`). The agent uses these tools directly; outputs are written to SQLite at call time.

**Architecture:** Single Node process: MCP server + background walker coordinator. The ctx-tree-backed tool handlers write to SQLite directly via `better-sqlite3` prepared statements. Walkers (`filter`, `staleness_marker`, `pruner`) run on `setInterval` with error boundaries and a startup sweep. No socket server, no hook, no threads in v1. (PostToolUse hook and socket ingestion are v1.1.)

**Tech Stack:** TypeScript 5, Bun (build + test + workspace), `better-sqlite3` (sync SQLite), `@modelcontextprotocol/sdk`, `ulid`, `tree-sitter` + language parsers, `micromatch` (glob matching), system dep: `rg ≥13` only. (`jq`, `socat`, `awk` are v1.1 hook deps, not needed in v1.)

export const DEFAULT_CONFIG: CtxTreeConfig = {
  embeddingModel: 'nomic-embed-text',
  summarizerModel: 'claude-haiku-4-5',
  retention: { staleHours: 24, supersededDays: 7 },
  walkers: {
    embeddingIdleMs: 5000,
    embeddingBatchSize: 32,
    summarizerSubtreeThreshold: 25,
    dedupeIntervalMs: 60000,
    stalenessIntervalMs: 30000,
    prunerIntervalMs: 300000,
  },
  capture: { maxBytes: 100000, filterMinSize: 50 },
};

function readJson(path: string): Partial<CtxTreeConfig> {
  try {
    return JSON.parse(readFileSync(path, 'utf8'));
  } catch {
    return {};
  }
}

export function loadConfig(projectRoot?: string): CtxTreeConfig {
  const globalPath = join(process.env.HOME ?? '~', '.ctx-tree', 'config.json');
  const globalOverride = existsSync(globalPath) ? readJson(globalPath) : {};

  const projectPath = projectRoot
    ? join(projectRoot, '.ctx-tree', 'config.json')
    : null;
  const projectOverride = projectPath && existsSync(projectPath)
    ? readJson(projectPath)
    : {};

  // Shallow merge: top-level keys replace wholesale (per spec §2)
  return { ...DEFAULT_CONFIG, ...globalOverride, ...projectOverride };
}
```

import { execSync } from 'child_process';
import { createHash } from 'crypto';

export function computeProjectHash(cwd: string): string {
  try {
    const gitRoot = execSync('git rev-parse --show-toplevel', {
      cwd,
      stdio: ['pipe', 'pipe', 'pipe'],
    }).toString().trim();
    const firstCommit = execSync('git rev-list --max-parents=0 HEAD', {
      cwd,
      stdio: ['pipe', 'pipe', 'pipe'],
    }).toString().trim();
    return createHash('sha256')
      .update(`${gitRoot}:${firstCommit}`)
      .digest('hex')
      .slice(0, 16);
  } catch {
    return createHash('sha256').update(cwd).digest('hex').slice(0, 16);
  }
}
```

> **v1.1 addition:** `resolveProjectHash`, `registerProject`, `deregisterProject`, and the `projects.tsv` management functions are added in v1.1 alongside the hook. In v1 the server only needs `computeProjectHash` to determine the DB directory.

import BetterSqlite3 from 'better-sqlite3';
import type Database from 'better-sqlite3';
import { mkdirSync } from 'fs';
import { dirname } from 'path';

export function openDb(dbPath: string): Database.Database {
  mkdirSync(dirname(dbPath), { recursive: true, mode: 0o700 });

  const db = new BetterSqlite3(dbPath, { verbose: undefined });
  db.pragma('journal_mode = WAL');
  db.pragma('foreign_keys = ON');
  db.pragma('synchronous = NORMAL');

  db.exec(`
    CREATE TABLE IF NOT EXISTS nodes (
      id            TEXT PRIMARY KEY,
      parent_id     TEXT REFERENCES nodes(id) ON DELETE SET NULL,
      kind          TEXT NOT NULL CHECK(kind IN ('session','file_chunk','tool_output','summary','note','observation')),
      source_uri    TEXT,
      content       TEXT NOT NULL DEFAULT '',
      content_hash  TEXT NOT NULL DEFAULT '',
      status        TEXT NOT NULL DEFAULT 'pending' CHECK(status IN ('pending','live','stale','superseded','pruned')),
      mtime         INTEGER NOT NULL DEFAULT 0,
      created_at    INTEGER NOT NULL,
      updated_at    INTEGER NOT NULL,
      truncated     INTEGER NOT NULL DEFAULT 0,
      original_bytes INTEGER NOT NULL DEFAULT 0,
      metadata      TEXT NOT NULL DEFAULT '{}'
    );

    CREATE INDEX IF NOT EXISTS idx_nodes_parent ON nodes(parent_id);
    CREATE INDEX IF NOT EXISTS idx_nodes_status ON nodes(status);
    CREATE INDEX IF NOT EXISTS idx_nodes_created ON nodes(created_at, parent_id);
    CREATE INDEX IF NOT EXISTS idx_nodes_source_uri ON nodes(source_uri);
    CREATE INDEX IF NOT EXISTS idx_nodes_content_hash ON nodes(content_hash);

    CREATE TABLE IF NOT EXISTS edges (
      src_id     TEXT NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
      dst_id     TEXT NOT NULL REFERENCES nodes(id) ON DELETE CASCADE,
      kind       TEXT NOT NULL CHECK(kind IN ('derived_from','references','summarizes','supersedes')),
      created_at INTEGER NOT NULL,
      PRIMARY KEY (src_id, dst_id, kind)
    );

    CREATE INDEX IF NOT EXISTS idx_edges_dst ON edges(dst_id);

    CREATE VIRTUAL TABLE IF NOT EXISTS nodes_fts USING fts5(
      id UNINDEXED,
      content,
      content='nodes',
      content_rowid='rowid'
    );

    CREATE TRIGGER IF NOT EXISTS nodes_fts_insert AFTER INSERT ON nodes BEGIN
      INSERT INTO nodes_fts(rowid, id, content) VALUES (new.rowid, new.id, new.content);
    END;

    CREATE TRIGGER IF NOT EXISTS nodes_fts_update AFTER UPDATE ON nodes BEGIN
      INSERT INTO nodes_fts(nodes_fts, rowid, id, content) VALUES ('delete', old.rowid, old.id, old.content);
      INSERT INTO nodes_fts(rowid, id, content) VALUES (new.rowid, new.id, new.content);
    END;

    CREATE TRIGGER IF NOT EXISTS nodes_fts_delete AFTER DELETE ON nodes BEGIN
      INSERT INTO nodes_fts(nodes_fts, rowid, id, content) VALUES ('delete', old.rowid, old.id, old.content);
    END;

    CREATE TABLE IF NOT EXISTS nodes_vec (
      id              TEXT PRIMARY KEY REFERENCES nodes(id) ON DELETE CASCADE,
      embedding       BLOB,
      embedding_model TEXT NOT NULL,
      embedding_dim   INTEGER NOT NULL,
      embedded_at     INTEGER NOT NULL
    );
  `);

  return db;
}

export function closeDb(db: Database.Database): void {
  try { db.close(); } catch { /* ignore */ }
}
```

import type Database from 'better-sqlite3';
import type { CtxTreeEdge, EdgeKind, CtxTreeNode } from './types';

export function insertEdge(db: Database.Database, edge: Omit<CtxTreeEdge, 'created_at'>): void {
  db.prepare(`
    INSERT OR IGNORE INTO edges (src_id, dst_id, kind, created_at) VALUES (?, ?, ?, ?)
  `).run(edge.src_id, edge.dst_id, edge.kind, Date.now());
}

export function getNeighbors(
  db: Database.Database,
  nodeId: string,
  edgeKinds?: EdgeKind[]
): CtxTreeNode[] {
  const kindFilter = edgeKinds?.length
    ? `AND e.kind IN (${edgeKinds.map(() => '?').join(',')})`
    : '';
  const params = edgeKinds?.length
    ? [nodeId, nodeId, ...edgeKinds]
    : [nodeId, nodeId];
  return db.prepare(`
    SELECT DISTINCT n.* FROM nodes n
    JOIN edges e ON (e.src_id = ? AND e.dst_id = n.id)
                 OR (e.dst_id = ? AND e.src_id = n.id)
    WHERE n.status = 'live' ${kindFilter}
  `).all(...params) as CtxTreeNode[];
}

export function getEdgesFrom(db: Database.Database, srcId: string): CtxTreeEdge[] {
  return db.prepare('SELECT * FROM edges WHERE src_id = ?').all(srcId) as CtxTreeEdge[];
}

import micromatch from 'micromatch';
import { BASH_DENY_COMMANDS, OUTPUT_REDACTION_PATTERNS } from './bash';
import { DEFAULT_PATH_DENY_GLOBS } from './paths';

export function shouldDropBashCommand(command: string, extraPatterns: RegExp[] = []): boolean {
  const all = [...BASH_DENY_COMMANDS, ...extraPatterns];
  return all.some(re => re.test(command));
}

export function redactBashOutput(output: string): string {
  let result = output;
  for (const { pattern, tag } of OUTPUT_REDACTION_PATTERNS) {
    result = result.replace(pattern, `[REDACTED:${tag}]`);
  }
  return result;
}

export function shouldDropPath(filePath: string, extraGlobs: string[] = []): boolean {
  const allGlobs = [...DEFAULT_PATH_DENY_GLOBS, ...extraGlobs];
  // Check basename against non-path patterns, full path against ** patterns
  const basename = filePath.split('/').pop() ?? '';
  return micromatch.isMatch(filePath, allGlobs) || micromatch.isMatch(basename, allGlobs);
}

import type Database from 'better-sqlite3';
import type { CtxTreeConfig } from '../store/types';
import { getPendingNodes, updateNodeStatus, getNodeByContentHash, pruneNode } from '../store/nodes';

const BATCH_SIZE = 50;

export function runFilterWalker(db: Database.Database, config: CtxTreeConfig): void {
  const pending = getPendingNodes(db, BATCH_SIZE);
  if (pending.length === 0) return;

  // Process the whole batch inside one transaction for ~50× throughput vs row-per-commit.
  const processBatch = db.transaction(() => {
    for (const node of pending) {
      if (node.original_bytes < config.capture.filterMinSize) {
        pruneNode(db, node.id);
        continue;
      }
      const existing = db.prepare(
        "SELECT id FROM nodes WHERE content_hash = ? AND status = 'live' AND id != ? LIMIT 1"
      ).get(node.content_hash, node.id);
      if (existing) {
        pruneNode(db, node.id);
        continue;
      }
      updateNodeStatus(db, node.id, 'live');
    }
  });
  processBatch();
}

import { insertEdge } from '../store/edges';

export function runStalenessWalker(db: Database.Database, _config: CtxTreeConfig): void {
  // Re-stat file_chunk nodes that haven't been checked recently
  const checkBefore = Date.now() - 5000; // recheck if not updated in last 5s
  const chunks = getLiveFileChunks(db, checkBefore);

  for (const node of chunks) {
    const meta = JSON.parse(node.metadata) as { filePath?: string };
    const filePath = meta.filePath ?? node.source_uri?.replace(/^file:\/\//, '').split('#')[0];
    if (!filePath) continue;

    try {
      const stat = statSync(filePath);
      const currentMtime = stat.mtimeMs;
      if (currentMtime !== node.mtime) {
        updateNodeStatus(db, node.id, 'stale');
      }
    } catch {
      // File gone — mark stale
      updateNodeStatus(db, node.id, 'stale');
    }
  }

export function runPrunerWalker(db: Database.Database, config: CtxTreeConfig): void {
  const now = Date.now();
  const staleThreshold = now - config.retention.staleHours * 60 * 60 * 1000;
  const supersededThreshold = now - config.retention.supersededDays * 24 * 60 * 60 * 1000;

  for (const node of getStaleNodes(db, staleThreshold)) {
    pruneNode(db, node.id);
  }
  for (const node of getSupersededNodes(db, supersededThreshold)) {
    pruneNode(db, node.id);
  }
}

  /** Call once at startup to process any pending rows left by a previous crashed session. */
  startupSweep(db: Database.Database, config: CtxTreeConfig): void {
    // Drain all pre-existing pending rows before starting interval walkers.
    let swept = 0;
    while (true) {
      const before = (db.prepare("SELECT COUNT(*) as n FROM nodes WHERE status='pending'").get() as { n: number }).n;
      if (before === 0) break;
      withErrorBoundary('filter-startup-sweep', () => runFilterWalker(db, config));
      swept++;
      if (swept > 1000) break; // safety cap — should never hit in practice
    }
    if (swept > 0) process.stderr.write(`ctx-tree: startup sweep processed pending rows in ${swept} passes\n`);
  }

  start(db: Database.Database, config: CtxTreeConfig): void {
    this.startupSweep(db, config);

    this.timers.push(
      setInterval(() => withErrorBoundary('filter', () => runFilterWalker(db, config)), 500),
      setInterval(() => withErrorBoundary('staleness', () => runStalenessWalker(db, config)), config.walkers.stalenessIntervalMs),
      setInterval(() => withErrorBoundary('pruner', () => runPrunerWalker(db, config)), config.walkers.prunerIntervalMs),
    );
  }

  stop(): void {
    this.timers.forEach(t => clearInterval(t));
    this.timers = [];
  }
}

export function buildFilterSQL(filters: Filters = {}): FilterSQL {
  const clauses: string[] = [];
  const params: (string | number)[] = [];

  const statuses = filters.status ?? ['live'];
  clauses.push(`status IN (${statuses.map(() => '?').join(',')})`);
  params.push(...statuses);

  if (filters.kind?.length) {
    clauses.push(`kind IN (${filters.kind.map(() => '?').join(',')})`);
    params.push(...filters.kind);
  }
  if (filters.since != null) { clauses.push('created_at >= ?'); params.push(filters.since); }
  if (filters.until != null) { clauses.push('created_at <= ?'); params.push(filters.until); }
  if (filters.parent_id)     { clauses.push('parent_id = ?'); params.push(filters.parent_id); }
  if (filters.session_id) {
    clauses.push("json_extract(metadata,'$.session_id') = ?");
    params.push(filters.session_id);
  }

  return { where: clauses.join(' AND '), params };
}

import type { CtxTreeNode, EdgeKind, Filters } from '../store/types';

const MAX_DEPTH = 5;

export function getNeighborsDeep(
  db: Database.Database,
  nodeId: string,
  depth = 1,
  edgeKinds?: EdgeKind[],
  filters: Filters = {}
): CtxTreeNode[] {
  const cap = Math.min(depth, MAX_DEPTH);
  const { where, params } = buildFilterSQL(filters);
  const kindFilter = edgeKinds?.length
    ? `AND e.kind IN (${edgeKinds.map(() => '?').join(',')})`
    : '';
  const kindParams = edgeKinds ?? [];

  const visited = new Set<string>([nodeId]);
  const result: CtxTreeNode[] = [];
  let frontier = [nodeId];

  for (let d = 0; d < cap; d++) {
    if (frontier.length === 0) break;
    const placeholders = frontier.map(() => '?').join(',');
    const next = db.prepare(`
      SELECT DISTINCT n.* FROM nodes n
      JOIN edges e ON (e.src_id IN (${placeholders}) AND e.dst_id = n.id)
                   OR (e.dst_id IN (${placeholders}) AND e.src_id = n.id)
      WHERE ${where} ${kindFilter}
    `).all(...frontier, ...frontier, ...params, ...kindParams) as CtxTreeNode[];

    const newNodes = next.filter(n => !visited.has(n.id));
    for (const n of newNodes) { visited.add(n.id); result.push(n); }
    frontier = newNodes.map(n => n.id);
  }

  return result;
}

```typescript
import { readFileSync, statSync } from 'fs';
import { createHash } from 'crypto';
import { ulid } from 'ulid';
import { extname } from 'path';
import type Database from 'better-sqlite3';
import type { CtxTreeConfig } from '../store/types';
import { insertNode, getNodeBySourceUri, updateNodeStatus } from '../store/nodes';
import { insertEdge } from '../store/edges';
import { shouldDropPath } from '../redaction';

export interface ReadParams {
  path: string;
  lines?: [number, number]; // [start, end] 1-based inclusive
  budget_tokens?: number;
}

export interface ReadResult {
  nodeId: string;
  content: string;
  truncated: boolean;
  chunking: 'treesitter' | 'window';
}

interface Chunk {
  startLine: number; // 1-based
  endLine: number;
  content: string;
  symbolName?: string;
}

function estimateTokens(text: string): number {
  return Math.ceil(text.length / 4);
}

function windowChunk(lines: string[], windowSize = 200): Chunk[] {
  const chunks: Chunk[] = [];
  for (let i = 0; i < lines.length; i += windowSize) {
    const slice = lines.slice(i, i + windowSize);
    chunks.push({ startLine: i + 1, endLine: i + slice.length, content: slice.join('\n') });
  }
  return chunks;
}

type TreeSitterLanguage = { [key: string]: unknown };

const LANG_MAP: Record<string, () => TreeSitterLanguage> = {
  '.ts': () => require('tree-sitter-typescript').typescript,
  '.tsx': () => require('tree-sitter-typescript').tsx,
  '.js': () => require('tree-sitter-javascript'),
  '.jsx': () => require('tree-sitter-javascript'),
  '.py': () => require('tree-sitter-python'),
  '.rs': () => require('tree-sitter-rust'),
  '.go': () => require('tree-sitter-go'),
  '.sh': () => require('tree-sitter-bash'),
};

function treeSitterChunk(source: string, lang: TreeSitterLanguage): Chunk[] {
  try {
    const Parser = require('tree-sitter');
    const parser = new Parser();
    parser.setLanguage(lang);
    const tree = parser.parse(source);
    const lines = source.split('\n');
    const chunks: Chunk[] = [];

    function walk(node: { type: string; startPosition: { row: number }; endPosition: { row: number }; childCount: number; children: unknown[]; text: string }) {
      const isTopLevel = [
        'function_declaration', 'class_declaration', 'method_definition',
        'lexical_declaration', 'variable_declaration', 'export_statement',
        'function_definition', 'class_definition',  // Python
        'function_item', 'struct_item', 'enum_item', 'impl_item',  // Rust
        'function_declaration',  // Go
      ].includes(node.type);

      if (isTopLevel) {
        const startLine = node.startPosition.row + 1;
        const endLine = node.endPosition.row + 1;
        const content = lines.slice(startLine - 1, endLine).join('\n');
        chunks.push({ startLine, endLine, content, symbolName: node.type });
        return; // don't recurse into top-level symbols
      }
      if (node.childCount > 0) {
        for (const child of node.children as typeof node[]) walk(child);
      }
    }

    walk(tree.rootNode);
    return chunks.length > 0 ? chunks : windowChunk(lines);
  } catch {
    return windowChunk(source.split('\n'));
  }
}

export async function ctxTreeRead(
  db: Database.Database,
  config: CtxTreeConfig,
  params: ReadParams
): Promise<ReadResult> {
  const { path, lines, budget_tokens = 2000 } = params;

  if (shouldDropPath(path, config.capture.pathDenylistExtra ?? [])) {
    throw new Error(`Path rejected by denylist: ${path}`);
  }

  const stat = statSync(path);
  const mtime = Math.round(stat.mtimeMs);
  const ext = extname(path).toLowerCase();
  const fileRootUri = `file://${path}`;

  // Check for cached file root node with matching mtime
  const cached = getNodeBySourceUri(db, fileRootUri);
  if (cached && cached.mtime === mtime) {
    const meta = JSON.parse(cached.metadata) as { chunking: 'treesitter' | 'window' };
    return { nodeId: cached.id, content: cached.content, truncated: cached.truncated === 1, chunking: meta.chunking };
  }

  // If cached but stale, mark stale and create supersedes edge
  if (cached) updateNodeStatus(db, cached.id, 'stale');

  // Read file
  let raw = readFileSync(path, 'utf8');
  const originalBytes = Buffer.byteLength(raw, 'utf8');
  let truncated = false;
  if (originalBytes > config.capture.maxBytes) {
    raw = raw.slice(0, config.capture.maxBytes);
    truncated = true;
  }

  const langLoader = LANG_MAP[ext];
  let chunks: Chunk[];
  let chunking: 'treesitter' | 'window';

  if (langLoader) {
    try {
      const lang = langLoader();
      chunks = treeSitterChunk(raw, lang);
      chunking = 'treesitter';
    } catch {
      chunks = windowChunk(raw.split('\n'));
      chunking = 'window';
    }
  } else {
    chunks = windowChunk(raw.split('\n'));
    chunking = 'window';
  }

  // Filter to requested line range if specified
  if (lines) {
    chunks = chunks.filter(c => c.startLine <= lines[1] && c.endLine >= lines[0]);
  }

  // Budget-pack chunks
  const budget = budget_tokens;
  let used = 0;
  const included: Chunk[] = [];
  for (const chunk of chunks) {
    const tokens = estimateTokens(chunk.content);
    if (used + tokens > budget) break;
    included.push(chunk);
    used += tokens;
  }

  const content = included.map(c => c.content).join('\n\n');
  const contentHash = createHash('sha256').update(content).digest('hex');
  const nodeId = ulid();

  insertNode(db, nodeId, {
    parent_id: null,
    kind: 'file_chunk',
    source_uri: fileRootUri,
    content,
    content_hash: contentHash,
    status: 'live',
    mtime,
    truncated: truncated ? 1 : 0,
    original_bytes: originalBytes,
    metadata: JSON.stringify({ filePath: path, chunking, chunkCount: included.length }),
  });

  // Write supersedes edge if we replaced a cached node
  if (cached) insertEdge(db, { src_id: nodeId, dst_id: cached.id, kind: 'supersedes' });

  return { nodeId, content, truncated, chunking };
}

export async function ctxTreeGrep(

  params: GrepParams

  const { pattern, path = '.', caseInsensitive, fileGlob, maxCount = 500 } = params;

  if (shouldDropPath(path, config.capture.pathDenylistExtra ?? [])) {

  if (fileGlob) args.push('--glob', fileGlob);
  args.push('--max-count', String(maxCount));
  args.push('--', pattern, path);

  let rawOutput = '';
  try {
    rawOutput = execSync(args.join(' '), {
      maxBuffer: config.capture.maxBytes,
      stdio: ['pipe', 'pipe', 'pipe'],
    }).toString('utf8');
  } catch (err: unknown) {
    // rg exits 1 for no matches — not a real error
    const e = err as { status?: number; stdout?: Buffer };
    if (e.status === 1) {
      rawOutput = e.stdout?.toString('utf8') ?? '';
    } else if ((e.status ?? 0) > 1) {
      throw new Error(`rg failed: ${err}`);
    }
  }

  const matches = rawOutput.split('\n').filter(Boolean);
  const content = matches.join('\n');
  const originalBytes = Buffer.byteLength(content, 'utf8');
  const truncContent = content.slice(0, config.capture.maxBytes);
  const contentHash = createHash('sha256').update(truncContent).digest('hex');
  const nodeId = ulid();

  if (content.length >= config.capture.filterMinSize) {
    insertNode(db, nodeId, {
      parent_id: null,
      kind: 'tool_output',
      source_uri: `tool:Grep#${contentHash.slice(0, 8)}`,
      content: truncContent,
      content_hash: contentHash,
      status: 'live',
      mtime: 0,
      truncated: originalBytes > config.capture.maxBytes ? 1 : 0,
      original_bytes: originalBytes,
      metadata: JSON.stringify({ tool: 'Grep', pattern, path }),
    });
  }

  return { nodeId, matches };
}

```typescript
import type Database from 'better-sqlite3';
import type { CtxTreeNode, ComposeManifest } from '../store/types';

export interface ComposeParams {
  node_ids: string[];
  budget_tokens: number;
  format?: 'raw' | 'outline' | 'mixed';
  query?: string;
  depth?: number;
}

export interface ComposeResult {
  content: string;
  manifest: ComposeManifest;
}

function estimateTokens(text: string): number {
  return Math.ceil(text.length / 4);
}

function scoreNode(
  node: CtxTreeNode,
  graphDistance: number,
  queryRank: number,
  hasQuery: boolean
): number {
  const wDist = hasQuery ? 0.6 : 0.7;
  const wRecency = 0.3;
  const wQuery = hasQuery ? 0.1 : 0.0;
  const ageHours = (Date.now() - node.updated_at) / (1000 * 60 * 60);
  const recencyDecay = Math.exp(-ageHours / 24);
  return wDist * (1 / (1 + graphDistance)) + wRecency * recencyDecay + wQuery * queryRank;
}

function expandGraph(
  db: Database.Database,
  seedIds: string[],
  maxDepth: number
): Map<string, number> {
  const visited = new Map<string, number>(); // id → min distance from any seed
  const queue: [string, number][] = seedIds.map(id => [id, 0]);

  while (queue.length > 0) {
    const [id, dist] = queue.shift()!;
    if (visited.has(id)) continue;
    visited.set(id, dist);
    if (dist >= maxDepth) continue;

    const children = db.prepare(
      "SELECT id FROM nodes WHERE parent_id = ? AND status = 'live'"
    ).all(id) as { id: string }[];

    const edgeNeighbors = db.prepare(`
      SELECT DISTINCT n.id FROM nodes n
      JOIN edges e ON (e.src_id = ? AND e.dst_id = n.id)
                   OR (e.dst_id = ? AND e.src_id = n.id)
      WHERE n.status = 'live'
    `).all(id, id) as { id: string }[];

    for (const { id: nextId } of [...children, ...edgeNeighbors]) {
      if (!visited.has(nextId)) queue.push([nextId, dist + 1]);
    }
  }

  return visited;
}

function getFtsRanks(db: Database.Database, query: string, ids: string[]): Map<string, number> {
  if (!query || ids.length === 0) return new Map();
  const placeholders = ids.map(() => '?').join(',');
  const rows = db.prepare(`
    SELECT n.id, rank FROM nodes n
    JOIN nodes_fts f ON f.id = n.id
    WHERE nodes_fts MATCH ? AND n.id IN (${placeholders})
    ORDER BY rank
  `).all(query, ...ids) as { id: string; rank: number }[];

  // Normalize ranks: lowest (most negative) rank = highest score
  if (rows.length === 0) return new Map();
  const minRank = Math.min(...rows.map(r => r.rank));
  const maxRank = Math.max(...rows.map(r => r.rank));
  const range = maxRank - minRank || 1;
  return new Map(rows.map(r => [r.id, (r.rank - maxRank) / range])); // 0..1, higher = better
}

function formatOutline(node: CtxTreeNode): string {
  const preview = node.content.slice(0, 120).replace(/\n/g, ' ');
  return `[${node.id}] ${node.kind}: ${preview}…`;
}

export async function ctxTreeCompose(
  db: Database.Database,
  params: ComposeParams
): Promise<ComposeResult> {
  const { node_ids, budget_tokens, format = 'raw', query, depth = 2 } = params;

  if (format === 'mixed') {
    throw new Error("format='mixed' is deferred to v1.1 — summary nodes not available in v1");
  }

  const distanceMap = expandGraph(db, node_ids, depth);
  if (distanceMap.size === 0) {
    return { content: '', manifest: { included: [], dropped: [], truncated: [] } };
  }

  const allIds = [...distanceMap.keys()];
  const candidates = db.prepare(
    `SELECT * FROM nodes WHERE id IN (${allIds.map(() => '?').join(',')}) AND status = 'live'`
  ).all(...allIds) as CtxTreeNode[];

  const ftsRanks = query ? getFtsRanks(db, query, allIds) : new Map<string, number>();
  const hasQuery = !!query;

  const scored = candidates.map(node => ({
    node,
    score: scoreNode(node, distanceMap.get(node.id) ?? 99, ftsRanks.get(node.id) ?? 0, hasQuery),
  })).sort((a, b) => b.score - a.score);

  const included: CtxTreeNode[] = [];
  const dropped: ComposeManifest['dropped'] = [];
  let usedTokens = 0;

  for (const { node } of scored) {
    const tokens = estimateTokens(format === 'outline' ? formatOutline(node) : node.content);
    if (usedTokens + tokens <= budget_tokens) {
      included.push(node);
      usedTokens += tokens;
    } else {
      dropped.push({ id: node.id, reason: 'over_budget' });
    }
  }

  // Also note any seed IDs that ended up excluded because they were superseded/pruned
  const liveIds = new Set(candidates.map(n => n.id));
  for (const id of node_ids) {
    if (!liveIds.has(id)) {
      const n = db.prepare('SELECT status FROM nodes WHERE id = ?').get(id) as { status: string } | undefined;
      if (n && (n.status === 'superseded' || n.status === 'pruned')) {
        dropped.push({ id, reason: n.status as 'superseded' | 'pruned' });
      }
    }
  }

  const truncated = included.filter(n => n.truncated === 1).map(n => n.id);

  let content: string;
  if (format === 'outline') {
    content = included.map(formatOutline).join('\n');
  } else {
    content = included.map(n => n.content).join('\n\n---\n\n');
  }

  return {
    content,
    manifest: { included: included.map(n => n.id), dropped, truncated },
  };
}
```

External, persistent, tree/graph-shaped context store for Claude Code sessions.

## What it does

Indexes files and search results in a local SQLite store when you use ctx-tree-backed MCP tools. Exposes tools for surgical recall — read a function body, search across sessions, compose a token-budget context bundle.

**v1.1:** Adds a PostToolUse hook for passive capture of native Read/Grep/Bash outputs.

## Install

```sh
/plugin install ctx-tree@agent-marketplace
```

## Requirements

- macOS arm64, Linux x64/arm64
- `rg ≥13` (`brew install ripgrep` / `apt install ripgrep`)

## Tools

| Tool | Description |
|------|-------------|
| `ctx_tree_read` | Read file with tree-sitter chunking + mtime cache |
| `ctx_tree_grep` | Ripgrep search, results stored |
| `ctx_tree_search` | FTS5 keyword search across store |
| `ctx_tree_compose` | Budget-bounded context bundle from seed nodes |
| `ctx_tree_neighbors` | Graph walk from node |
| `ctx_tree_path_to_root` | Walk parent chain to root |
| `ctx_tree_recent` | Recently captured nodes |
```

```typescript
import { execSync } from 'child_process';
import { existsSync, mkdirSync, writeFileSync, readFileSync, createWriteStream } from 'fs';
import { join } from 'path';
import { createHash } from 'crypto';
import https from 'https';

const PLUGIN_ROOT = process.env.CLAUDE_PLUGIN_ROOT ?? join(__dirname, '..', '..');
const PACKAGE_JSON = JSON.parse(readFileSync(join(PLUGIN_ROOT, 'mcp', 'package.json'), 'utf8'));
const VERSION = PACKAGE_JSON.version ?? 'latest';
const REPO = 'joeblackwaslike/ctx-tree';

const platform = process.platform;
const arch = process.arch;
const TRIPLE = `${platform}-${arch}`;

const SUPPORTED = ['darwin-arm64', 'linux-x64', 'linux-arm64'];
if (!SUPPORTED.includes(TRIPLE)) {
  process.stderr.write(`ctx-tree: unsupported platform ${TRIPLE}\n`);
  process.exit(1);
}

const NATIVE_DIR = join(PLUGIN_ROOT, 'mcp', 'native', TRIPLE);
const STAMP = join(NATIVE_DIR, '.installed');

if (existsSync(STAMP)) {
  process.stdout.write(`ctx-tree: native deps already installed for ${TRIPLE}\n`);
  process.exit(0);
}

mkdirSync(NATIVE_DIR, { recursive: true });

const ARCHIVE_NAME = `ctx-tree-native-${TRIPLE}.tar.gz`;
const CHECKSUM_NAME = `${ARCHIVE_NAME}.sha256`;
const BASE_URL = `https://github.com/${REPO}/releases/download/v${VERSION}`;

function download(url: string, dest: string): Promise<void> {
  return new Promise((resolve, reject) => {
    const file = createWriteStream(dest);
    https.get(url, (res) => {
      if (res.statusCode === 302 || res.statusCode === 301) {
        file.close();
        download(res.headers.location!, dest).then(resolve).catch(reject);
        return;
      }
      res.pipe(file);
      file.on('finish', () => file.close(() => resolve()));
    }).on('error', reject);
  });
}

async function main() {
  process.stdout.write(`ctx-tree: downloading native deps for ${TRIPLE} (v${VERSION})...\n`);

  const archivePath = join(NATIVE_DIR, ARCHIVE_NAME);
  const checksumPath = join(NATIVE_DIR, CHECKSUM_NAME);

  await download(`${BASE_URL}/${ARCHIVE_NAME}`, archivePath);
  await download(`${BASE_URL}/${CHECKSUM_NAME}`, checksumPath);

  // Verify checksum
  const expected = readFileSync(checksumPath, 'utf8').trim().split(/\s+/)[0];
  const actual = createHash('sha256').update(readFileSync(archivePath)).digest('hex');
  if (actual !== expected) {
    process.stderr.write(`ctx-tree: checksum mismatch for ${ARCHIVE_NAME}\n`);
    process.stderr.write(`  expected: ${expected}\n  actual:   ${actual}\n`);
    process.exit(1);
  }

  // Extract
  execSync(`tar -xzf ${archivePath} -C ${NATIVE_DIR}`, { stdio: 'inherit' });

  writeFileSync(STAMP, new Date().toISOString());
  process.stdout.write(`ctx-tree: native deps installed to ${NATIVE_DIR}\n`);
}

main().catch(e => {
  process.stderr.write(`ctx-tree postinstall failed: ${e}\n`);
  process.exit(1);
});
```

```sh
#!/bin/sh
# Prepare release artifacts: build server, bundle native deps per platform.
# Usage: ./prepare-release.sh <version>
set -eu

VERSION="${1:?Usage: prepare-release.sh <version>}"
REPO="joeblackwaslike/ctx-tree"
TRIPLES="darwin-arm64 linux-x64 linux-arm64"
DIST="$(pwd)/release-artifacts"
mkdir -p "$DIST"

echo "Building server bundle..."
bun build --cwd mcp src/server.ts --target node --outdir dist \
  --external better-sqlite3 \
  --external tree-sitter \
  --external tree-sitter-typescript \
  --external tree-sitter-javascript \
  --external tree-sitter-python \
  --external tree-sitter-rust \
  --external tree-sitter-go \
  --external tree-sitter-bash
cd ..

for TRIPLE in $TRIPLES; do
  echo "Fetching native deps for $TRIPLE..."
  OUTDIR="$DIST/native-$TRIPLE"
  mkdir -p "$OUTDIR"

  # better-sqlite3 prebuilt (via prebuild-install)
  PLATFORM=$(echo "$TRIPLE" | cut -d- -f1)
  ARCH=$(echo "$TRIPLE" | cut -d- -f2)

  # sqlite-vec: grab from its GitHub release
  SQLITE_VEC_VERSION="v0.1.6"
  case "$TRIPLE" in
    darwin-arm64) SQLITE_VEC_FILE="sqlite-vec-${SQLITE_VEC_VERSION#v}-macos-aarch64.tar.gz" ;;
    linux-x64)    SQLITE_VEC_FILE="sqlite-vec-${SQLITE_VEC_VERSION#v}-linux-x86_64.tar.gz" ;;
    linux-arm64)  SQLITE_VEC_FILE="sqlite-vec-${SQLITE_VEC_VERSION#v}-linux-aarch64.tar.gz" ;;
  esac

  gh release download "$SQLITE_VEC_VERSION" \
    --repo asg017/sqlite-vec \
    --pattern "$SQLITE_VEC_FILE" \
    --dir "$OUTDIR" || echo "Warning: sqlite-vec download failed for $TRIPLE"

  # Bundle into archive
  ARCHIVE="$DIST/ctx-tree-native-$TRIPLE.tar.gz"
  tar -czf "$ARCHIVE" -C "$OUTDIR" .
  sha256sum "$ARCHIVE" | awk '{print $1}' > "$ARCHIVE.sha256"
  echo "Created $ARCHIVE"
done

echo "Uploading to GitHub release v$VERSION..."
gh release create "v$VERSION" "$DIST"/ctx-tree-native-*.tar.gz "$DIST"/ctx-tree-native-*.sha256 \
  --repo "$REPO" \
  --title "v$VERSION" \
  --generate-notes
```

```typescript
import type Database from 'better-sqlite3';
import { ctxTreeRead } from '../../src/tools/read';
import { shouldDropPath, shouldDropBashCommand, redactBashOutput } from '../../src/redaction';
import type { CtxTreeConfig } from '../../src/store/types';
import { DEFAULT_CONFIG } from '../../src/config';

export interface SecurityResult {
  name: string;
  passed: boolean;
  error?: string;
}

const LEAKY_BASH_PAYLOADS = [
  { command: 'env', output: 'HOME=/Users/joe\nANTHROPIC_API_KEY=sk-ant-abc123secret\n' },
  { command: 'printenv', output: 'AWS_SECRET_ACCESS_KEY=wJalrXUtnFEMI/K7MDENG/bPxRfiCYEXAMPLEKEY\n' },
  { command: 'echo $TOKEN', output: 'AKIAIOSFODNN7EXAMPLE\n' },
  { command: 'cat .env', output: 'OPENAI_API_KEY=sk-abcdefghij1234567890abcdefghij1234567890\n' },
  { command: 'curl -H auth', output: 'Authorization: Bearer eyJhbGciOiJIUzI1NiJ9.eyJzdWIiOiJ1In0.abc_sig\n' },
];

const SECRET_FILE_PATHS = [
  '/home/user/.env',
  '/project/.env.local',
  '/home/user/.aws/credentials',
  '/home/user/.ssh/id_rsa',
  '/certs/server.pem',
  '/keys/service-account.json',
];

export async function runSecurityCases(db: Database.Database): Promise<SecurityResult[]> {
  const cfg: CtxTreeConfig = { ...DEFAULT_CONFIG };
  const results: SecurityResult[] = [];

  // Bash commands containing secrets must be flagged by shouldDropBashCommand;
  // outputs must have secrets redacted by redactBashOutput
  for (const { command, output } of LEAKY_BASH_PAYLOADS) {
    const dropped = shouldDropBashCommand(command);
    const redacted = redactBashOutput(output);
    const secretsGone = !redacted.includes('sk-') && !redacted.includes('AKIAIO') &&
      !redacted.includes('wJalrX') && !redacted.includes('eyJhbG');
    const passed = dropped || secretsGone;
    results.push({ name: `bash-redact:${command}`, passed,
      error: passed ? undefined : `Secret leaked in redacted output` });
  }

  // Secret file paths must be rejected by ctxTreeRead (throws "Path rejected by denylist")
  for (const filePath of SECRET_FILE_PATHS) {
    let passed = false;
    let error: string | undefined;
    try {
      await ctxTreeRead(db, cfg, { path: filePath, budget_tokens: 200 });
      error = `Expected rejection for ${filePath} but got a result`;
    } catch (e: any) {
      passed = String(e.message).includes('Path rejected by denylist');
      if (!passed) error = `Wrong error: ${e.message}`;
    }
    results.push({ name: `path-drop:${filePath}`, passed, error });
  }

  return results;
}
```

```typescript
import type Database from 'better-sqlite3';
import { ctxTreeRead } from '../../src/tools/read';
import type { CtxTreeConfig } from '../../src/store/types';

export interface LatencyResult {
  name: string;
  p50Ms: number;
  p95Ms: number;
  thresholdMs: number;
  passed: boolean;
}

async function measureMs(fn: () => Promise<void>, n: number): Promise<{ p50: number; p95: number }> {
  const times: number[] = [];
  for (let i = 0; i < n; i++) {
    const start = performance.now();
    await fn();
    times.push(performance.now() - start);
  }
  times.sort((a, b) => a - b);
  return { p50: times[Math.floor(n * 0.5)], p95: times[Math.floor(n * 0.95)] };
}

export async function runLatencyCases(
  db: Database.Database,
  config: CtxTreeConfig,
  fixturePath: string
): Promise<LatencyResult[]> {
  // Warm up cache — read once
  await ctxTreeRead(db, config, { path: fixturePath, budget_tokens: 2000 });

  // Measure cached reads (mtime unchanged → SQLite hit)
  const { p50, p95 } = await measureMs(
    () => ctxTreeRead(db, config, { path: fixturePath, budget_tokens: 2000 }),
    20
  );

  return [{
    name: 'ctx-tree.read cached p50',
    p50Ms: p50, p95Ms: p95,
    thresholdMs: 50,
    passed: p50 < 50,
  }];
}
```

```typescript
import { unlinkSync, existsSync, mkdirSync } from 'fs';
import { join } from 'path';
import { openDb, closeDb } from '../src/store/db';
import { DEFAULT_CONFIG } from '../src/config';
import { runSecurityCases } from './cases/security';
import { runLatencyCases } from './cases/latency';

const DB_PATH = '/tmp/ctx-tree-eval.db';
const FIXTURE_DIR = join(__dirname, 'fixtures', 'src');
const FIXTURE_FILE = join(FIXTURE_DIR, 'math.ts');

async function main() {
  // Clean state
  for (const f of [DB_PATH, DB_PATH + '-wal', DB_PATH + '-shm'])
    if (existsSync(f)) unlinkSync(f);

  const db = openDb(DB_PATH);
  const cfg = DEFAULT_CONFIG;
  let failed = 0;

  // ── Security ────────────────────────────────────────────────────────────────
  console.log('\n=== Security Cases ===');
  const secResults = await runSecurityCases(db);
  for (const r of secResults) {
    const icon = r.passed ? '✓' : '✗';
    console.log(`  ${icon} ${r.name}${r.error ? `: ${r.error}` : ''}`);
    if (!r.passed) failed++;
  }

  // ── Latency ─────────────────────────────────────────────────────────────────
  console.log('\n=== Latency Cases ===');
  const latResults = await runLatencyCases(db, cfg, FIXTURE_FILE);
  for (const r of latResults) {
    const icon = r.passed ? '✓' : '✗';
    console.log(`  ${icon} ${r.name}: p50=${r.p50Ms.toFixed(1)}ms p95=${r.p95Ms.toFixed(1)}ms (threshold: ${r.thresholdMs}ms)`);
    if (!r.passed) failed++;
  }

  // ── Schema rejection ────────────────────────────────────────────────────────
  console.log('\n=== Schema Rejection ===');
  const { ctxTreeCompose } = await import('../src/tools/compose');
  try {
    await ctxTreeCompose(db, { node_ids: ['x'], budget_tokens: 100, format: 'mixed' });
    console.log('  ✗ compose format=mixed should have thrown');
    failed++;
  } catch (e: unknown) {
    const msg = e instanceof Error ? e.message : String(e);
    if (msg.includes('deferred to v1.1')) {
      console.log('  ✓ compose format=mixed correctly rejected');
    } else {
      console.log(`  ✗ compose format=mixed threw wrong error: ${msg}`);
      failed++;
    }
  }

  closeDb(db);
  for (const f of [DB_PATH, DB_PATH + '-wal', DB_PATH + '-shm'])
    if (existsSync(f)) unlinkSync(f);

  console.log(`\n${failed === 0 ? '✓ All eval cases passed' : `✗ ${failed} case(s) failed`}\n`);
  process.exit(failed > 0 ? 1 : 0);
}

main().catch(e => { console.error(e); process.exit(1); });
```
